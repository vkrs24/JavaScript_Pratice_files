      4.Shorthand property, Destructuring, Shorthand method
      
      obj1 = {
        prop1 : value1,
        prop2 : value2
      };                             //Destructuring
      const {prop1, prop2} = obj1;
      console.log(prop1);
      console.log(prop2);

      obj2 = {
        prop1, //from obj1           //Shorthand property
        method() {
          //code
          ...                       //Shorthand method
          ...
          console.log('method');
        }
      };
      console.log(obj2);  //{prop1 : value1, method : f}
      obj2.method();
